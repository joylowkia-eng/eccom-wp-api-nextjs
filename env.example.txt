# WooCommerce API Configuration
# Replace these values with your actual WooCommerce credentials

# Your WordPress/WooCommerce site URL (without trailing slash)
NEXT_PUBLIC_WOOCOMMERCE_URL=https://your-wordpress-site.com

# WooCommerce REST API Consumer Key
NEXT_PUBLIC_WOOCOMMERCE_KEY=ck_your_consumer_key_here

# WooCommerce REST API Consumer Secret
NEXT_PUBLIC_WOOCOMMERCE_SECRET=cs_your_consumer_secret_here

# Instructions:
# 1. Go to your WordPress admin panel
# 2. Navigate to WooCommerce > Settings > Advanced > REST API
# 3. Click "Add key" to generate new API credentials
# 4. Set permissions to "Read/Write"
# 5. Copy the Consumer Key and Consumer Secret
# 6. Replace the placeholder values above with your actual credentials
# 7. Rename this file to .env.local
